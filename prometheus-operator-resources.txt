apiVersion: monitoring.coreos.com/v1
kind: PrometheusRule
metadata:
  name: sa-omf-alerts
  namespace: monitoring
  labels:
    app: sa-omf-collector
    prometheus: prometheus
    role: alert-rules
spec:
  groups:
  - name: sa-omf.rules
    rules:
    - alert: SAOMFHighCPUUsage
      expr: rate(otelcol_process_cpu_seconds_total[5m]) * 1000 > 300
      for: 5m
      labels:
        severity: warning
      annotations:
        summary: "SA-OMF collector high CPU usage"
        description: "SA-OMF collector is using more than 300 mCores of CPU for 5 minutes."
    
    - alert: SAOMFHighMemoryUsage
      expr: otelcol_process_memory_rss / 1024 / 1024 > 300
      for: 5m
      labels:
        severity: warning
      annotations:
        summary: "SA-OMF collector high memory usage"
        description: "SA-OMF collector is using more than 300 MiB of memory for 5 minutes."
    
    - alert: SAOMFLowCoverageScore
      expr: aemf_impact_adaptive_topk_coverage_score < 0.8
      for: 15m
      labels:
        severity: warning
      annotations:
        summary: "SA-OMF coverage score too low"
        description: "Coverage score has been below 0.8 for 15 minutes. Consider increasing K value."
    
    - alert: SAOMFSafeModeActive
      expr: aemf_anom_safe_mode_active > 0
      for: 1m
      labels:
        severity: warning
      annotations:
        summary: "SA-OMF safe mode active"
        description: "SA-OMF collector is in safe mode. Check resources and configuration."
    
    - alert: SAOMFConfigChangeRate
      expr: rate(aemf_decision_patch_applied_total[5m]) > 0.5
      for: 15m
      labels:
        severity: warning
      annotations:
        summary: "SA-OMF high configuration change rate"
        description: "SA-OMF collector is making configuration changes at a high rate. Check for oscillations."
